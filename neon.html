JavaScript

$('.neonDiv').hover(function(){
  //Mouse in
  window.flicker = true;
  flick();
}, function(){
  //Mouse out
  window.flicker = false;
});

function flick(){
  if(window.flicker){

  //get a couple flicker duractions to choose from so it's more flickery
  longFlick = Math.floor((Math.random()*800)+20)/2;
  shortFlick = Math.floor((Math.random()*100)+1)/2;

  rand = Math.random();
    setTimeout(function(){
      rand = Math.random();
      $('.breakMe').addClass('broken');
      setTimeout(function(){
        $('.breakMe').removeClass('broken');
        flick();
      },rand > .5 ? longFlick : shortFlick);
    },rand > .5 ? longFlick : shortFlick);
  
  }
}
